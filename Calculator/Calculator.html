555     public String getOperand2(String text) {
556         String result = "";
557         char tmp[] = text.toCharArray();
558         int pos = 0;
559         for (int i = 0; i < text.length(); i++) {
560             if (tmp[i] == '+' || tmp[i] == '-' || tmp[i] == '*' || tmp[i] == '/' || tmp[i] == '^') {
561                 pos = i + 1;
562                 break;
563             }
564         }
565         result = String.copyValueOf(tmp, pos, text.length() - pos);
566         if (fstSigned) {
567             fstSigned = false;
568             result = getOperand2(result);
569         }
570         return result;
571     }

573     public String calculate() {
574         String result = "0";
575         System.out.println("Operand1: " + operand1);
576         System.out.println("Operator: " + operator);
577         System.out.println("Operand2: " + operand2);
578         if (operator.equals("+")) {
579             result = Float.toString(Float.parseFloat(operand1) + Float.parseFloat(operand2));
580         }
581         if (operator.equals("-")) {
582             result = Float.toString(Float.parseFloat(operand1) - Float.parseFloat(operand2));
583         }
584         if (operator.equals("*")) {
585             result = Float.toString(Float.parseFloat(operand1) * Float.parseFloat(operand2));
586         }
587         if (operator.equals("/")) {
588             result = Float.toString(Float.parseFloat(operand1) / Float.parseFloat(operand2));
589         }
590         if (operator.equals("^")) {
591             try {
592                 result = Double.toString(Math.pow(Double.parseDouble(operand1), Double.parseDouble(operand2)));
593             } catch (Exception e) {
594                 clearAll();
595                 screen.setText("Math Error");
596             }
597         }
598         System.out.println("Result: " + result);
599         if (Float.parseFloat(result) < 0) {
600             fstSigned = true;
601         } else {
602             fstSigned = false;
603         }
604         afterResult = true;
605         return result;
606     }

608     public void clickOperators(String operator) {
609         if (!initial) {
610             if (!operatorClicked) {
611                 if (!fstOptClicked) {
612                     operand1 = screen.getText();
613                     screen.setText(screen.getText() + operator);
614                     fstOptClicked = true;
615                 } else {
616                     operand2 = getOperand2(screen.getText());
617                     operand1 = calculate();
618                     screen.setText(operand1 + operator);
619                 }
620                 this.operator = operator;
621                 operatorClicked = true;
622                 dotClicked = false;
623             } else {
624                 if (!signedOp) {
625                     if (operator.equals("+") || operator.equals("-")) {
626                         screen.setText(screen.getText() + operator);
627                         signedOp = true;
628                     }
629                 }
630             }
631             afterResult = false;
632         } else {
633             if ((operator.equals("+") || operator.equals("-")) && !fstSigned) {
634                 screen.setText(operator);
635                 fstSigned = true;
636                 afterResult = false;
637             }
638         }
639     }
