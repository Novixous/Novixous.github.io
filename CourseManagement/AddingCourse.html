118     private void btnAddingActionPerformed(java.awt.event.ActionEvent evt) {                                          
119         CourseList courseList = new CourseList();
120         boolean check = false;
121         if (isEmpty(txtCode) || isEmpty(txtName) || isEmpty(txtCredit)) {
122             JOptionPane.showMessageDialog(this, "Please fill full the form");
123         } else {
124             try {
125                 check = courseList.addNewCourse(txtCode.getText(), txtName.getText(),
126                         Integer.parseInt(txtCredit.getText()));
127                 if (check) {
128                     JOptionPane.showMessageDialog(this, "Addding successfully");
129                 } else {
130                     JOptionPane.showMessageDialog(this, "Addding failed");
131                 }
132             } catch (SQLException ex) {
133                 JOptionPane.showMessageDialog(this, "We can't duplicate code");
134             } catch (NumberFormatException e) {
135                 JOptionPane.showMessageDialog(this, "Credit should be a number");
136             } catch (Exception e){
137                 e.printStackTrace();
138             }
139         }
140     }
