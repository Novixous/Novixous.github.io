119     private void btnSearchActionPerformed(java.awt.event.ActionEvent evt) {                                          
120         CourseList courseList = new CourseList();
121         Course course = courseList.searchByCourse(txtSearching.getText());
122         if (course != null) {
123             lblName.setText(course.getNameCourse());
124             lblCredit.setText(Integer.toString(course.getCredit()));
125         } else {
126             JOptionPane.showMessageDialog(this, "We dont't have that code");
127         }
128     }
