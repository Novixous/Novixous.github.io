24     public boolean addNewCourse(String code, String name, int credit) throws SQLException, NumberFormatException{
25         boolean check = false;
26         String sql = "Insert into Course Values(?,?,?)";
27         try (Connection conn = MyConnection.openConnection();
28                 PreparedStatement ps = conn.prepareStatement(sql);) {
29             ps.setString(1, code);
30             ps.setString(2, name);
31             ps.setInt(3, credit);
32             check = ps.executeUpdate() > 0;
33         }
34         return check;
35     }
36 
37     public List<Course> displayCourse() {
38         List<Course> result = new ArrayList<>();
39         String sql = "Select * From Course";
40         try (Connection conn = MyConnection.openConnection();
41                 PreparedStatement ps = conn.prepareStatement(sql);) {
42             ResultSet rs = ps.executeQuery();
43             while (rs.next()) {
44                 result.add(new Course(rs.getString("Code"),
45                         rs.getString("Name"), rs.getInt("Credit")));
46             }
47         } catch (Exception e) {
48             e.printStackTrace();
49         }
50         return result;
51     }
52 
53     public Course searchByCourse(String code) {
54         Course course = null;
55         String sql = "Select * From Course Where code = ?";
56         try (Connection conn = MyConnection.openConnection();
57                 PreparedStatement ps = conn.prepareStatement(sql);) {
58             ps.setString(1, code);
59             ResultSet rs = ps.executeQuery();
60             if (rs.next()) {
61                 course = new Course(rs.getString("Code"),
62                         rs.getString("Name"), rs.getInt("Credit"));
63             }
64         } catch (Exception e) {
65             e.printStackTrace();
66         }
67         return course;
68     }
